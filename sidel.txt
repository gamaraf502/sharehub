import java.util.Scanner;
public class Main
{
  public static void main(String[] args){
    Scanner sc = new Scanner (System.in);
    int n=sc.nextInt();
    int isPrime;
    for(int i=2;i<n;i++){
      isPrime=0;
      for(int j=3;j<i;j++){
        
        if(i%j==0)
        {
          isPrime=1;
          break;
        }
        }
        if(i==4)
        continue;
        if (isPrime==0)
        System.out.print(i+" ");
      
      sc.close();
    }
  }
}
