#!/bin/sh
echo "enter date:"
read d m y

# check month range
if [$m -lt 1] || [$m -gt 12]
then
        echo "invalid date"
        exit
fi

#proper leap year check
if[$((y%400)) -eq 0] || [$((y%4)) eq 0 -a $((y%100)) -ne 0]
then
        leap=1
else
        leap=0
fi

#set maximum days for each month
case $m in
        1|3|5|7|8|10|12
                max=31 ;;
        4|6|9|11)
                max=30 ;;
        2)
                if[$leap -eq 1]
                then
                        max=29
                else
                        max=28
                fi ;;
esac

#validate day
if[$d -ge 1] && [$d -le $max]
then
        echo "valid date"
else
        echo "invalid date"
fi
